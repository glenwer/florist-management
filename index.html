const SHEET_NAME = '주문 기록'; const HEADERS = [ '주문 일자', '손님 이름', '손님 전화번호', '손님 주소', '크레딧 카드 번호', '받는 사람 이름', '받는 사람 전화번호', '받는 사람 주소', '배달 일자', '상품 내용', '상품 금액', '배달료', '텍스 비율', '텍스 금액', '총액' ]; function doPost(e) { let result = { result: 'error', message: 'Unknown error' }; try { if (e.postData && e.postData.type === "application/json") { const data = JSON.parse(e.postData.contents); const action = data.action; if (action === 'register') { result = handleRegistration(data); } else if (action === 'update') { result = handleUpdate(data); } else { result.message = 'Invalid action specified.'; } } else { result.message = 'Invalid data type or missing postData.'; } } catch (error) { Logger.log('doPost Error: ' + error.toString()); result.message = 'Server processing error: ' + error.toString(); } // ✅ 수정: add1e → addHeader return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON) .addHeader('Access-Control-Allow-Origin', '*'); } function doGet(e) { let result = { result: 'error', message: 'Unknown error' }; try { const customerName = e.parameter.customerName; if (customerName) { result = handleSearch(customerName); } else { result.result = 'not found'; result.message = 'Customer name parameter is missing.'; } } catch (error) { Logger.log('doGet Error: ' + error.toString()); result.message = 'Server processing error: ' + error.toString(); } // ✅ 수정: add1e → addHeader return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON) .addHeader('Access-Control-Allow-Origin', '*'); } function handleRegistration(data) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); if (!sheet) { return { result: 'error', message: 'Sheet named ' + SHEET_NAME + ' not found.' }; } const row = HEADERS.map(header => { return data[header] !== undefined ? data[header] : ''; }); sheet.appendRow(row); Logger.log('Registration Success: ' + data['손님 이름'] + ' on ' + data['주문 일자']); return { result: 'success' }; } function handleUpdate(data) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const searchKey = data.searchKey; if (!sheet || !searchKey) { return { result: 'error', message: 'Sheet not found or search key missing.' }; } const [targetName, targetDate] = searchKey.split('|||'); const range = sheet.getDataRange(); const values = range.getValues(); const headers = values[0]; let updateRowIndex = -1; const nameIndex = headers.indexOf('손님 이름'); const dateIndex = headers.indexOf('주문 일자'); if (nameIndex === -1 || dateIndex === -1) { return { result: 'error', message: 'Missing "손님 이름" or "주문 일자" header in sheet.' }; } for (let i = 1; i < values.length; i++) { const sheetDate = values[i][dateIndex]; const formattedSheetDate = sheetDate instanceof Date ? Utilities.formatDate(sheetDate, Session.getScriptTimeZone(), 'yyyy-MM-dd') : sheetDate; if (values[i][nameIndex] === targetName && formattedSheetDate === targetDate) { updateRowIndex = i + 1; break; } } if (updateRowIndex !== -1) { const newRow = HEADERS.map(header => data[header] !== undefined ? data[header] : ''); sheet.getRange(updateRowIndex, 1, 1, HEADERS.length).setValues([newRow]); Logger.log('Update Success: ' + targetName + ' on row ' + updateRowIndex); return { result: 'success' }; } else { return { result: 'error', message: 'Record not found for update.' }; } } function handleSearch(customerName) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); if (!sheet) { return { result: 'error', message: 'Sheet named ' + SHEET_NAME + ' not found.' }; } const range = sheet.getDataRange(); const values = range.getValues(); const headers = values[0]; const nameIndex = headers.indexOf('손님 이름'); if (nameIndex === -1) { return { result: 'error', message: 'Missing "손님 이름" header in sheet.' }; } // ✅ 수정: 미사용 변수 latestRecord, latestOrderDate 제거 for (let i = values.length - 1; i >= 1; i--) { const row = values[i]; if (row[nameIndex] === customerName) { const record = {}; headers.forEach((header, index) => { let value = row[index]; if (value instanceof Date) { value = Utilities.formatDate(value, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } record[header] = value; }); return { result: 'success', data: record }; } } return { result: 'not found', message: 'No matching record found.' }; }
