/** * Apps Script의 메인 함수입니다. 웹 앱에서 GET 또는 POST 요청을 처리합니다. * 웹 앱 배포 시 '실행 사용자: 나', '액세스 권한: 누구나'로 설정해야 합니다. * * 중요: 이 코드를 사용하기 전에 스프레드시트의 탭(시트 이름)이 '주문 기록'인지 확인하십시오. */ // 사용할 스프레드시트 이름과 시트 이름 설정 const SHEET_NAME = '주문 기록'; // 모든 입력 필드의 헤더 순서를 정의합니다. 이 순서대로 시트에 기록됩니다. const HEADERS = [ '주문 일자', '손님 이름', '손님 전화번호', '손님 주소', '크레딧 카드 번호', '받는 사람 이름', '받는 사람 전화번호', '받는 사람 주소', '배달 일자', '상품 내용', '상품 금액', '배달료', '텍스 비율', // 이 값은 입력 폼에서 계산에 사용되지만, 기록을 위해 저장합니다. '텍스 금액', '총액' ]; /** * 웹 앱의 POST 요청을 처리합니다 (데이터 등록/수정). * @param {Object} e - 요청 이벤트 객체 */ function doPost(e) { let result = { result: 'error', message: 'Unknown error' }; try { // 1. JSON 데이터 파싱: 클라이언트에서 JSON.stringify로 전송했으므로 이 방식으로 읽어야 합니다. if (e.postData && e.postData.type === "application/json") { const data = JSON.parse(e.postData.contents); const action = data.action; if (action === 'register') { result = handleRegistration(data); } else if (action === 'update') { result = handleUpdate(data); } else { result.message = 'Invalid action specified.'; } } else { result.message = 'Invalid data type or missing postData.'; } } catch (error) { // 에러 발생 시 로그 기록 Logger.log('doPost Error: ' + error.toString()); result.message = 'Server processing error: ' + error.toString(); } // CORS 및 Content-Type 설정 return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON) .add1e('Access-Control-Allow-Origin', '*'); } /** * 웹 앱의 GET 요청을 처리합니다 (데이터 검색). * @param {Object} e - 요청 이벤트 객체 */ function doGet(e) { let result = { result: 'error', message: 'Unknown error' }; try { const customerName = e.parameter.customerName; if (customerName) { result = handleSearch(customerName); } else { result.result = 'not found'; result.message = 'Customer name parameter is missing.'; } } catch (error) { Logger.log('doGet Error: ' + error.toString()); result.message = 'Server processing error: ' + error.toString(); } // CORS 및 Content-Type 설정 return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON) .add1e('Access-Control-Allow-Origin', '*'); } /** * 새 주문을 시트에 등록합니다. * @param {Object} data - 클라이언트에서 전송된 JSON 데이터 */ function handleRegistration(data) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); if (!sheet) { return { result: 'error', message: 'Sheet named ' + SHEET_NAME + ' not found.' }; } // 헤더 순서에 맞춰서 데이터 배열을 생성합니다. const row = HEADERS.map(header => { // 데이터에 해당 헤더가 없으면 공백으로 처리 (예: 텍스 금액, 총액 등은 0.00으로 들어옴) return data[header] !== undefined ? data[header] : ''; }); sheet.appendRow(row); // 성공 시 로그 기록 Logger.log('Registration Success: ' + data['손님 이름'] + ' on ' + data['주문 일자']); return { result: 'success' }; } /** * 기존 주문 기록을 업데이트합니다. * @param {Object} data - 클라이언트에서 전송된 JSON 데이터 */ function handleUpdate(data) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const searchKey = data.searchKey; // "손님 이름|||주문 일자" 형식 if (!sheet || !searchKey) { return { result: 'error', message: 'Sheet not found or search key missing.' }; } const [targetName, targetDate] = searchKey.split('|||'); const range = sheet.getDataRange(); const values = range.getValues(); const headers = values[0]; // 첫 번째 행은 헤더입니다. let updateRowIndex = -1; // 헤더에서 '손님 이름'과 '주문 일자'의 인덱스를 찾습니다. const nameIndex = headers.indexOf('손님 이름'); const dateIndex = headers.indexOf('주문 일자'); if (nameIndex === -1 || dateIndex === -1) { return { result: 'error', message: 'Missing "손님 이름" or "주문 일자" header in sheet.' }; } // 2번째 행부터 순회하며 일치하는 레코드를 찾습니다. for (let i = 1; i < values.length; i++) { // 시트의 주문 일자는 Date 객체로 읽히므로 문자열로 변환하여 비교합니다. const sheetDate = values[i][dateIndex]; const formattedSheetDate = sheetDate instanceof Date ? Utilities.formatDate(sheetDate, Session.getScriptTimeZone(), 'yyyy-MM-dd') : sheetDate; if (values[i][nameIndex] === targetName && formattedSheetDate === targetDate) { updateRowIndex = i + 1; // Apps Script 행 인덱스는 1부터 시작 break; } } if (updateRowIndex !== -1) { // 업데이트할 새 데이터 배열 생성 (HEADER 순서와 동일하게) const newRow = HEADERS.map(header => data[header] !== undefined ? data[header] : ''); // 시트에 데이터 쓰기 (업데이트) sheet.getRange(updateRowIndex, 1, 1, HEADERS.length).setValues([newRow]); Logger.log('Update Success: ' + targetName + ' on row ' + updateRowIndex); return { result: 'success' }; } else { return { result: 'error', message: 'Record not found for update.' }; } } /** * 손님 이름으로 가장 최근 주문 기록을 검색합니다. * @param {string} customerName - 검색할 손님 이름 */ function handleSearch(customerName) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); if (!sheet) { return { result: 'error', message: 'Sheet named ' + SHEET_NAME + ' not found.' }; } const range = sheet.getDataRange(); const values = range.getValues(); const headers = values[0]; const nameIndex = headers.indexOf('손님 이름'); if (nameIndex === -1) { return { result: 'error', message: 'Missing "손님 이름" header in sheet.' }; } let latestRecord = null; let latestOrderDate = new Date(0); // 2번째 행부터 역순으로 순회 (최신 기록을 먼저 찾기 위해 역순 순회로 변경) for (let i = values.length - 1; i >= 1; i--) { const row = values[i]; if (row[nameIndex] === customerName) { // 행 데이터를 객체로 변환 const record = {}; headers.forEach((header, index) => { let value = row[index]; // 날짜 포맷팅 처리 (웹 폼의 input[type=date]에 맞춰 yyyy-MM-dd 형식으로 변환) if (value instanceof Date) { value = Utilities.formatDate(value, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } record[header] = value; }); // 최신 레코드만 반환하므로, 첫 번째로 찾은 일치하는 레코드를 반환하고 종료합니다. return { result: 'success', data: record }; } } // 아무것도 찾지 못했을 경우 return { result: 'not found', message: 'No matching record found.' }; }
